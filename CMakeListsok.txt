cmake_minimum_required(VERSION 3.16.3) 
project(sir VERSION 0.1.0)

set(CMAKE_CXX_COMPILER g++) #versione del compilatore
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD_REQUIRED ON)  
set(CMAKE_CXX_STANDARD 17) # standard del linguaggio c++, versione 17

if(CMAKE_COMPILER_IS_GNUCXX)
    message(STATUS "GCC detected, adding compile flags")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++17 -pedantic -Wall -Wextra")
endif(CMAKE_COMPILER_IS_GNUCXX)

add_executable(sir src/SirModel.cpp src/SirData.cpp src/SirModelExtended.cpp src/SirManageConfig.cpp src/sfml/plot.cpp  src/sfml/area.cpp src/sfml/axis.cpp src/sfml/curve.cpp src/sfml/grid.cpp  src/sfml/line.cpp main.cpp)
target_include_directories(sir PUBLIC "include/" "include/sfml")
TARGET_LINK_LIBRARIES(${PROJECT_NAME}  sfml-window sfml-system sfml-graphics)
